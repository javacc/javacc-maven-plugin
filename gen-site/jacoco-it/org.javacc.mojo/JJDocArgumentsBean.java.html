package org.javacc.mojo;

import java.util.List;

import org.apache.maven.project.MavenProject;

/**
 * The bean handling JJDoc options as command line arguments.
 *
 * @since 3.8.0
 * @author Maͫzͣaͬsͨ
 */
public class JJDocArgumentsBean extends AbstractArgumentsBean {
  
  /**
   * The list of single full command line arguments.
   *
   * @see JJDocGoalMojo#jjdocCmdLineArgs
   */
  protected List<String> jjdocCmdLineArgs;
  
  @Override
  protected List<String> getProcCmdLineArgs() {
    return jjdocCmdLineArgs;
  }
  
  @Override
  protected void setProcCmdLineArgs(final List<String> argsList) {
    jjdocCmdLineArgs = argsList;
  }
  
  /** The JJDoc default output sub directory. */
  protected static final String defOutSubDir = "generated-jjdoc";
  
  /** The array of corresponding default output sub directories. */
  protected static final String[] defOutSubDirs = new String[] {
      defOutSubDir
  };
  
  @Override
  protected void findSpecificOptions(final MavenProject project, final String param)
      throws PluginException {
    super.findSpecificOptions(project, param, null, defOutSubDirs);
  }
}
